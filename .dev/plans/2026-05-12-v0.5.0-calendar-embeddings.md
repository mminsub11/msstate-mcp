# v0.5.0 Calendar Hybrid Retrieval Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Land hybrid BM25 + dense-vector retrieval (Voyage `voyage-3-lite`, RRF-fused) for calendar queries in both surfaces (Cloudflare Worker + stdio plugin), with graceful BM25-only fallback when `VOYAGE_API_KEY` is unset or the embedding call fails. Ship as v0.5.0.

**Architecture:** Vectors are baked offline by `scripts/build-worker-corpus.mjs` into `worker/corpus.json` (full) and `msstate-policies/dist/calendar-vectors.json` (sidecar). At query time, both surfaces embed the user query via Voyage REST, do in-memory cosine similarity against the baked vectors, fuse the result with BM25 ranks via Reciprocal Rank Fusion (k=60), then apply the existing v0.4.1 smart-fallback layer. If the embed call returns null (no key / failure / circuit-breaker tripped), the system falls back to the v0.4.1 BM25-only path.

**Tech Stack:** TypeScript (strict), `node:test` + `tsx`, `node:crypto` for SHA-256, native `fetch` for Voyage REST, esbuild. Spec at `.dev/specs/2026-05-12-v0.5.0-calendar-embeddings-design.md`.

import { createHash } from "node:crypto";

/** SHA-256 hex of a canonical pipe-joined event identity. Stable across
 *  builds; missing fields normalize to empty strings so an undefined `term`
 *  and an empty-string `term` produce the same hash. Used as the vector
 *  lookup key in the embedding sidecar. */
export function contentHash(row: {
  event: string;
  term?: string;
  description?: string;
}): string {
  const canon = `${row.event}|${row.term ?? ""}|${row.description ?? ""}`;
  return createHash("sha256").update(canon, "utf8").digest("hex");
}
```

  let dot = 0, na = 0, nb = 0;

/** Per-vector symmetric int8 quantization. Cosine similarity is invariant
 *  to scale, so we only need the absolute-max for rescaling. Roundtrip
 *  cosine error is < 1e-3 on 512-dim random vectors. */

export interface QuantizedVec {
  /** Per-vector scale: original ≈ data[i] * scale. 0 when all input is 0. */
  scale: number;
  /** int8 values in [-127, 127]. */
  data: Int8Array;
}

export function quantizeInt8(v: Float32Array): QuantizedVec {
  let max = 0;
  for (let i = 0; i < v.length; i++) {
    const abs = Math.abs(v[i]);
    if (abs > max) max = abs;
  }
  if (max === 0) {
    return { scale: 0, data: new Int8Array(v.length) };
  }
  const scale = max / 127;
  const data = new Int8Array(v.length);
  for (let i = 0; i < v.length; i++) {
    const q = Math.round(v[i] / scale);
    data[i] = q < -127 ? -127 : q > 127 ? 127 : q;
  }
  return { scale, data };
}

export function dequantize(q: QuantizedVec): Float32Array {
  const out = new Float32Array(q.data.length);
  for (let i = 0; i < q.data.length; i++) {
    out[i] = q.data[i] * q.scale;
  }
  return out;
}
```

/** Reciprocal Rank Fusion: score(d) = Σ_L 1 / (k + rank_L(d)), summed
 *  over each list L that contains d. k=60 is the Cormack et al. canonical
 *  value. Returns IDs sorted by score desc, deduplicated, capped at limit. */
export function reciprocalRankFusion(
  rankedLists: readonly (readonly string[])[],
  k = 60,
  limit = 10,
): string[] {
  const scores = new Map<string, number>();
  for (const list of rankedLists) {
    for (let i = 0; i < list.length; i++) {
      const rank = i + 1;
      const id = list[i];
      scores.set(id, (scores.get(id) ?? 0) + 1 / (k + rank));
    }
  }
  return [...scores.entries()]
    .sort((a, b) => b[1] - a[1])
    .slice(0, limit)
    .map(([id]) => id);
}
```

 *
 * - Reads VOYAGE_API_KEY from process.env at module load.
 * - 2s timeout per call.
 * - LRU cache (size 256, keyed by raw text).

 * - Logs structured errors to stderr (never echoes err.message to clients).
 */
import { log } from "../log.js";
import { EMBED_DIM } from "./types.js";

const VOYAGE_URL = "https://api.voyageai.com/v1/embeddings" as const;
const VOYAGE_MODEL = "voyage-3-lite" as const;
const TIMEOUT_MS = 2000;
const LRU_SIZE = 256;

const apiKey = process.env.VOYAGE_API_KEY;
export const isEmbeddingAvailable = typeof apiKey === "string" && apiKey.length > 0;

if (!isEmbeddingAvailable) {
  log("warn", "VOYAGE_API_KEY unset; calendar tool will fall back to BM25-only retrieval");
}

// Circuit breaker: 5 failures in 30s → trip for 60s.
let recentFailures: number[] = [];
let trippedUntil = 0;
const TRIP_THRESHOLD = 5;
const TRIP_WINDOW_MS = 30_000;
const TRIP_DURATION_MS = 60_000;

function noteFailure(): void {
  const now = Date.now();
  recentFailures = recentFailures.filter((t) => now - t < TRIP_WINDOW_MS);
  recentFailures.push(now);
  if (recentFailures.length >= TRIP_THRESHOLD) {
    trippedUntil = now + TRIP_DURATION_MS;
    recentFailures = [];
    log("warn", "embedding circuit breaker tripped", { until: new Date(trippedUntil).toISOString() });
  }
}

function isTripped(): boolean {
  return Date.now() < trippedUntil;
}

// Simple LRU: Map preserves insertion order, evict from front on overflow.
const lru = new Map<string, Float32Array>();

function lruGet(key: string): Float32Array | undefined {
  const v = lru.get(key);
  if (v !== undefined) {
    lru.delete(key);
    lru.set(key, v);
  }
  return v;
}

function lruSet(key: string, v: Float32Array): void {
  if (lru.has(key)) lru.delete(key);
  lru.set(key, v);
  if (lru.size > LRU_SIZE) {
    const first = lru.keys().next().value as string;
    lru.delete(first);
  }
}

/**
 * Embed a single query string. Returns null when:
 *   - VOYAGE_API_KEY is unset
 *   - circuit breaker is tripped
 *   - the network call fails (timeout, non-200, malformed response)
 * Never throws. Callers must handle null by falling back to BM25.
 */
export async function embedQuery(text: string): Promise<Float32Array | null> {
  if (!isEmbeddingAvailable) return null;
  if (isTripped()) return null;
  const cached = lruGet(text);
  if (cached) return cached;

  const start = Date.now();
  try {
    const res = await fetch(VOYAGE_URL, {
      method: "POST",
      headers: {
        "Authorization": `Bearer ${apiKey}`,
        "Content-Type": "application/json",
      },
      body: JSON.stringify({
        model: VOYAGE_MODEL,
        input: [text],
        input_type: "query",
      }),
      signal: AbortSignal.timeout(TIMEOUT_MS),
    });

    if (!res.ok) {
      log("warn", "voyage non-200", { status: res.status, duration_ms: Date.now() - start });
      noteFailure();
      return null;
    }

    const json = (await res.json()) as { data?: Array<{ embedding?: number[] }> };
    const arr = json.data?.[0]?.embedding;
    if (!Array.isArray(arr) || arr.length !== EMBED_DIM) {
      log("warn", "voyage malformed response", { len: arr?.length, duration_ms: Date.now() - start });
      noteFailure();
      return null;
    }

    const vec = new Float32Array(arr);
    lruSet(text, vec);
    return vec;
  } catch (err) {
    log("warn", "voyage call failed", {
      err_class: err instanceof Error ? err.constructor.name : "unknown",
      duration_ms: Date.now() - start,
    });
    noteFailure();
    return null;
  }
}

/**
 * Batch-embed multiple texts. Used by the build script, NOT at query time.
 * Throws on failure so the build can abort and refuse to ship a poisoned corpus.
 */
export async function embedBatch(
  texts: string[],
  inputType: "document" | "query" = "document",
): Promise<Float32Array[]> {
  if (!isEmbeddingAvailable) {
    throw new Error("VOYAGE_API_KEY unset; cannot embed at build time");
  }
  const res = await fetch(VOYAGE_URL, {
    method: "POST",
    headers: {
      "Authorization": `Bearer ${apiKey}`,
      "Content-Type": "application/json",
    },
    body: JSON.stringify({ model: VOYAGE_MODEL, input: texts, input_type: inputType }),
    signal: AbortSignal.timeout(30_000),
  });
  if (!res.ok) {
    throw new Error(`voyage batch failed: status=${res.status}`);
  }
  const json = (await res.json()) as { data: Array<{ embedding: number[] }> };
  return json.data.map((d) => new Float32Array(d.embedding));
}

import { createHash } from "node:crypto";

// ---- Embedding constants (mirror src/calendars/types.ts) -----------------
const VOYAGE_URL = "https://api.voyageai.com/v1/embeddings";
const EMBED_MODEL = "voyage-3-lite";
const EMBED_DIM = 512;
const BATCH_SIZE = 128;
const MAX_PER_BATCH_RETRIES = 3;

/** SHA-256 hex of canonical event identity. Must match src/calendars/hash.ts. */
function contentHash(row) {
  const canon = `${row.event}|${row.term ?? ""}|${row.description ?? ""}`;
  return createHash("sha256").update(canon, "utf8").digest("hex");
}

/** Render the text input passed to Voyage for one row. Capped at 1024 chars
 *  (Voyage handles up to 32k tokens; capping prevents pathological PDFs). */
function embedText(row) {
  const parts = [row.event];
  if (row.term) parts.push(` — ${row.term}`);
  if (row.description) parts.push(`. ${row.description}`);
  return parts.join("").slice(0, 1024);
}

/** Symmetric int8 quantization per vector. */
function quantizeInt8(arr) {
  let max = 0;
  for (const x of arr) {
    const abs = Math.abs(x);
    if (abs > max) max = abs;
  }
  if (max === 0) return { scale: 0, data: new Int8Array(arr.length) };
  const scale = max / 127;
  const data = new Int8Array(arr.length);
  for (let i = 0; i < arr.length; i++) {
    const q = Math.round(arr[i] / scale);
    data[i] = q < -127 ? -127 : q > 127 ? 127 : q;
  }
  return { scale, data };
}

async function voyageBatchWithRetry(texts, apiKey, attempt = 1) {
  try {
    const res = await fetch(VOYAGE_URL, {
      method: "POST",
      headers: { "Authorization": `Bearer ${apiKey}`, "Content-Type": "application/json" },
      body: JSON.stringify({ model: EMBED_MODEL, input: texts, input_type: "document" }),
      signal: AbortSignal.timeout(30_000),
    });
    if (!res.ok) throw new Error(`status=${res.status}`);
    const json = await res.json();
    return json.data.map((d) => d.embedding);
  } catch (err) {
    if (attempt >= MAX_PER_BATCH_RETRIES) throw err;
    const backoff = [1000, 3000, 10000][attempt - 1] ?? 10000;
    console.error(`[embed] batch failed (attempt ${attempt}): ${err.message}; retrying in ${backoff}ms`);
    await new Promise((r) => setTimeout(r, backoff));
    return voyageBatchWithRetry(texts, apiKey, attempt + 1);
  }
}

async function embedRowsAndQuantize(rows, existingVecMap, apiKey) {
  for (const r of rows) r.contentHash = contentHash(r);
  const newRows = rows.filter((r) => !existingVecMap[r.contentHash]);
  console.error(`[embed] ${rows.length} rows total; ${newRows.length} need fresh embeddings; ${rows.length - newRows.length} reuse existing`);

  const newVecMap = {};
  let failedCount = 0;
  for (let i = 0; i < newRows.length; i += BATCH_SIZE) {
    const batch = newRows.slice(i, i + BATCH_SIZE);
    try {
      const vecs = await voyageBatchWithRetry(batch.map(embedText), apiKey);
      for (let j = 0; j < batch.length; j++) {
        const vec = vecs[j];
        if (!Array.isArray(vec) || vec.length !== EMBED_DIM) {
          failedCount++;
          continue;
        }
        const q = quantizeInt8(vec);
        newVecMap[batch[j].contentHash] = {
          scale: q.scale,
          data: Buffer.from(q.data.buffer).toString("base64"),
        };
      }
    } catch (err) {
      failedCount += batch.length;
      console.error(`[embed] batch ${i}-${i + batch.length} permanently failed: ${err.message}`);
    }
  }

  const failureRate = newRows.length > 0 ? failedCount / newRows.length : 0;
  if (failureRate > 0.1) {
    throw new Error(`refusing to ship a poisoned calendar corpus — embedding failure rate ${(failureRate * 100).toFixed(1)}% exceeds 10%`);
  }

  return { ...existingVecMap, ...newVecMap };
}

/** Ranked cosine results. Skips rows without an indexed vector. */
export function cosineSearchRanked(qVec: Float32Array, topK = 20): Array<{ rowId: string; rank: number }> {
  if (qVec.length === 0 || vecMapModule.size === 0) return [];
  let qNorm = 0;
  for (let i = 0; i < qVec.length; i++) qNorm += qVec[i] * qVec[i];
  qNorm = Math.sqrt(qNorm);
  if (qNorm === 0) return [];

  const scored: Array<{ rowId: string; score: number }> = [];
  for (const d of docs) {
    const hash = d.row.contentHash;
    if (!hash) continue;
    const q = vecMapModule.get(hash);
    if (!q) continue;
    const dvec = dequantize(q);
    let dot = 0, dn = 0;
    for (let i = 0; i < qVec.length; i++) {
      dot += qVec[i] * dvec[i];
      dn += dvec[i] * dvec[i];
    }
    const cos = dot / (qNorm * Math.sqrt(dn) || 1);
    scored.push({ rowId: hash, score: cos });
  }
  scored.sort((a, b) => b.score - a.score);
  return scored.slice(0, topK).map((s, i) => ({ rowId: s.rowId, rank: i + 1 }));

let embedRecentFailures: number[] = [];

function embedNoteFailure(): void {
  const now = Date.now();
  embedRecentFailures = embedRecentFailures.filter((t) => now - t < 30_000);
  embedRecentFailures.push(now);
  if (embedRecentFailures.length >= 5) {
    embedTrippedUntil = now + 60_000;
    embedRecentFailures = [];
  }
}

async function embedQuery(text: string, apiKey: string | undefined): Promise<Float32Array | null> {
  if (!apiKey) return null;
  if (Date.now() < embedTrippedUntil) return null;
  const cached = embedLru.get(text);
  if (cached) {
    embedLru.delete(text);
    embedLru.set(text, cached);
    return cached;
  }
  try {
    const res = await fetch(VOYAGE_URL, {
      method: "POST",
      headers: { "Authorization": `Bearer ${apiKey}`, "Content-Type": "application/json" },
      body: JSON.stringify({ model: VOYAGE_MODEL, input: [text], input_type: "query" }),
      signal: AbortSignal.timeout(EMBED_TIMEOUT_MS),
    });
    if (!res.ok) { embedNoteFailure(); return null; }
    const json = (await res.json()) as { data?: Array<{ embedding?: number[] }> };
    const arr = json.data?.[0]?.embedding;
    if (!Array.isArray(arr) || arr.length !== CAL_EMBED_DIM) {
      embedNoteFailure();
      return null;
    }
    const v = new Float32Array(arr);
    embedLru.set(text, v);
    if (embedLru.size > EMBED_LRU_SIZE) {
      const first = embedLru.keys().next().value as string;
      embedLru.delete(first);
    }
    return v;
  } catch {
    embedNoteFailure();
    return null;
  }
}

function dequantize(q: DecodedEmbedding): Float32Array {
  const out = new Float32Array(q.data.length);
  for (let i = 0; i < q.data.length; i++) out[i] = q.data[i] * q.scale;

}

  qNorm = Math.sqrt(qNorm);

/**
 * Hybrid retrieval eval — runs both BM25-only and hybrid modes against
 * the live indexed corpus and reports top-3 recall per bucket.
 *
 * Run: VOYAGE_API_KEY=... npx tsx evals/calendar-hybrid-eval.ts

import { loadAllCalendarRows } from "../src/calendars/corpus.js";
import {
  indexCalendarRows,
  searchCalendarRowsHybrid,

import { embedQuery } from "../src/calendars/embed.js";

interface EvalCase {
  q: string;
  expected_event_prefixes: string[]; // top-3 recall: any one of these in matches[0..2]
}

const SEMANTIC_GAP: EvalCase[] = [

  { q: "commencement december 2026", expected_event_prefixes: ["Commencement"] },

  { q: "classes begin spring 2026", expected_event_prefixes: ["Classes begin"] },
  { q: "independence day holiday", expected_event_prefixes: ["Independence Day"] },
  { q: "memorial day holiday", expected_event_prefixes: ["Memorial Day"] },
  { q: "deadline for payment in full", expected_event_prefixes: ["Deadline for payment"] },
];

const SMART_FALLBACK: EvalCase[] = [
  { q: "grad student spring 2027 schedule", expected_event_prefixes: ["Classes begin"] },
  { q: "housing event spring 2026 academic dates", expected_event_prefixes: ["Classes begin", "Spring Break"] },
  { q: "financial aid fall 2026 important dates", expected_event_prefixes: ["Classes begin", "Last day"] },
  { q: "exam schedule spring 2027 graduate", expected_event_prefixes: ["Classes begin", "Final Exam"] },
  { q: "tuition deadline fall 2026 housing", expected_event_prefixes: ["Deadline for payment", "Classes begin"] },
];

async function runOne(c: EvalCase, mode: "hybrid" | "bm25_only"): Promise<boolean> {
  const qVec = mode === "hybrid" ? await embedQuery(c.q) : null;
  const hits = searchCalendarRowsHybrid(c.q, 10, qVec);
  const top3 = hits.slice(0, 3).map((h) => h.row.event);
  return c.expected_event_prefixes.some((prefix) =>
    top3.some((event) => event.startsWith(prefix)),
  );
}

async function evalBucket(cases: EvalCase[], mode: "hybrid" | "bm25_only"): Promise<number> {
  let hits = 0;
  for (const c of cases) if (await runOne(c, mode)) hits++;
  return hits;
}

async function main() {
  const { rows, vecMap } = await loadAllCalendarRows();
  indexCalendarRows(rows, vecMap);

  const buckets: Array<[string, EvalCase[]]> = [
    ["semantic_gap", SEMANTIC_GAP],
    ["bm25_favorable", BM25_FAVORABLE],
    ["smart_fallback", SMART_FALLBACK],
  ];

  const results: Record<string, { bm25: number; hybrid: number; total: number }> = {};
  for (const [name, cases] of buckets) {
    const bm25 = await evalBucket(cases, "bm25_only");
    const hybrid = await evalBucket(cases, "hybrid");
    results[name] = { bm25, hybrid, total: cases.length };
  }

  console.log("Bucket            BM25-only    Hybrid      Δ");
  for (const [name, r] of Object.entries(results)) {
    const dPct = ((r.hybrid - r.bm25) / r.total) * 100;
    const sign = dPct >= 0 ? "+" : "";
    console.log(
      `${name.padEnd(18)} ${r.bm25}/${r.total}       ${r.hybrid}/${r.total}       ${sign}${dPct.toFixed(1)}pp`,
    );
  }

  const sg = results.semantic_gap;
  const bf = results.bm25_favorable;
  const liftPct = ((sg.hybrid - sg.bm25) / sg.total) * 100;
  const regressionPct = ((bf.bm25 - bf.hybrid) / bf.total) * 100;
  let exitCode = 0;
  if (liftPct < 10) {
    console.error(`\nFAIL: semantic_gap lift ${liftPct.toFixed(1)}pp < 10pp ship-blocker threshold`);
    exitCode = 1;
  }
  if (regressionPct > 5) {
    console.error(`\nFAIL: bm25_favorable regression ${regressionPct.toFixed(1)}pp > 5pp ship-blocker threshold`);
    exitCode = 1;
  }
  if (exitCode === 0) console.log("\nPASS: all bucket targets met");
  process.exit(exitCode);
}

main().catch((err) => {
  console.error("eval crashed:", err);
  process.exit(2);
});
```
